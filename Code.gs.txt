function ss_() { return SpreadsheetApp.getActiveSpreadsheet(); }

function sheet_(name) {
  const ss = ss_();
  let sh = ss.getSheetByName(name);

/** Ensures the Questions sheet has a 'Revealed' column, for spreadsheets created before this feature existed. */
function ensureQuestionColumns_() {

  const sh =
    sheet_(SHEET.QUESTIONS);


  let headers =
    sh
      .getRange(
        1,
        1,
        1,
        Math.max(
          sh.getLastColumn(),
          1
        )
      )
      .getValues()[0];


  // Add Department before Date
  if (
    headers.indexOf("Dept") === -1
  ) {

    const dateCol =
      headers.indexOf("Date") + 1;


    sh.insertColumnBefore(dateCol);


    sh
      .getRange(
        1,
        dateCol
      )
      .setValue("Dept");

  }


  // Refresh headers
  headers =
    sh
      .getRange(
        1,
        1,
        1,
        sh.getLastColumn()
      )
      .getValues()[0];


  // Add Year before Date
  if (
    headers.indexOf("Year") === -1
  ) {

    const dateCol =
      headers.indexOf("Date") + 1;


    sh.insertColumnBefore(dateCol);


    sh
      .getRange(
        1,
        dateCol
      )
      .setValue("Year");

  }


  // Refresh headers
  headers =
    sh
      .getRange(
        1,
        1,
        1,
        sh.getLastColumn()
      )
      .getValues()[0];


  // Ensure Revealed column
  if (
    headers.indexOf("Revealed") === -1
  ) {

    sh
      .getRange(
        1,
        sh.getLastColumn() + 1
      )
      .setValue("Revealed");

  }

}

/**********************************************************************
 * HELPERS
 **********************************************************************/
function sheetToObjects_(sh) {
  const values = sh.getDataRange().getValues();
  if (values.length < 2) return [];
  const headers = values.shift();
  return values.map(row => {
    const obj = {};
    headers.forEach((h, i) => obj[h] = row[i]);
    return obj;
  });
}

function fmtDate_(d) {
  if (typeof d === 'string') return d;
  return Utilities.formatDate(new Date(d), Session.getScriptTimeZone() || 'Asia/Kolkata', 'yyyy-MM-dd');
}

function todayStr_() {
  return Utilities.formatDate(new Date(), Session.getScriptTimeZone() || 'Asia/Kolkata', 'yyyy-MM-dd');
}

function isTestAvailableToday_() {

  const tz =
    Session.getScriptTimeZone() ||
    'Asia/Kolkata';

  const now =
    new Date();

  const currentTime =
    Utilities.formatDate(
      now,
      tz,
      'HH:mm'
    );

  // Full-day test availability
  return (
    currentTime >= '00:00' &&
    currentTime <= '23:49'
  );

}

function genPassword_(rollNo) {
  const rand = Math.floor(100 + Math.random() * 900);
  return String(rollNo).trim() + '@' + rand;
}

function newId_(prefix) {
  return prefix + Utilities.getUuid().split('-')[0].toUpperCase();
}

/**********************************************************************
 * STREAKS & ACTIVITY CALENDAR
 **********************************************************************/
function addDays_(dateStr, n) {
  const d = new Date(dateStr);
  d.setDate(d.getDate() + n);
  return Utilities.formatDate(d, Session.getScriptTimeZone() || 'Asia/Kolkata', 'yyyy-MM-dd');
}

function isNextDay_(d1, d2) {
  const diff = (new Date(d2) - new Date(d1)) / (1000 * 60 * 60 * 24);
  return diff === 1;
}

/** Current daily streak (consecutive days with at least one attempt) + best-ever streak. */
function getStudentStreak(rollNo) {
  const responses = sheetToObjects_(sheet_(SHEET.RESPONSES))
    .filter(r => String(r.RollNo).trim().toLowerCase() === String(rollNo).trim().toLowerCase());
  const dateSet = new Set(responses.map(r => fmtDate_(r.Date)));
  const dates = Array.from(dateSet).sort();
  if (dates.length === 0) return { current: 0, best: 0, totalActiveDays: 0 };

  let best = 1, run = 1;
  for (let i = 1; i < dates.length; i++) {
    run = isNextDay_(dates[i - 1], dates[i]) ? run + 1 : 1;
    if (run > best) best = run;
  }

  const today = todayStr_();
  let current = 0;
  let cursor = dateSet.has(today) ? today : addDays_(today, -1); // grace: ok if today just hasn't happened yet
  while (dateSet.has(cursor)) {
    current++;
    cursor = addDays_(cursor, -1);
  }
  return { current: current, best: best, totalActiveDays: dates.length };
}

/** Per-day attempt/correct/marks breakdown for a given month, for the activity calendar. */
function getMonthlyActivity(rollNo, year, month) {
  const now = new Date();
  const y = year || now.getFullYear();
  const m = month || (now.getMonth() + 1);
  const tz = Session.getScriptTimeZone() || 'Asia/Kolkata';

  const responses = sheetToObjects_(sheet_(SHEET.RESPONSES))
    .filter(r => String(r.RollNo).trim().toLowerCase() === String(rollNo).trim().toLowerCase());
  const map = {};
  responses.forEach(r => {
    const d = fmtDate_(r.Date);
    const dy = new Date(d);
    if (dy.getFullYear() === y && (dy.getMonth() + 1) === m) {
      if (!map[d]) map[d] = { attempted: 0, correct: 0, marks: 0 };
      map[d].attempted++;
      if (r.IsCorrect === true || r.IsCorrect === 'TRUE') map[d].correct++;
      map[d].marks += Number(r.Marks || 0);
    }
  });

  const daysInMonth = new Date(y, m, 0).getDate();
  const days = [];
  for (let day = 1; day <= daysInMonth; day++) {
    const dateStr = Utilities.formatDate(new Date(y, m - 1, day), tz, 'yyyy-MM-dd');
    const info = map[dateStr] || { attempted: 0, correct: 0, marks: 0 };
    days.push({ date: dateStr, day: day, attempted: info.attempted, correct: info.correct, marks: info.marks });
  }
  return { year: y, month: m, days: days };
}

/**********************************************************************
 * AUTH
 **********************************************************************/
function adminLogin(username, password) {
  const rows = sheetToObjects_(sheet_(SHEET.ADMIN));
  const found = rows.find(r => String(r.Username).trim() === String(username).trim()
    && String(r.Password).trim() === String(password).trim());
  if (!found) return { success: false, message: 'Invalid admin credentials' };
  return { success: true, role: 'admin', username: found.Username };
}

function studentLogin(rollNo, password) {
  const rows = sheetToObjects_(sheet_(SHEET.STUDENTS));
  const found = rows.find(r => String(r.RollNo).trim().toLowerCase() === String(rollNo).trim().toLowerCase()
    && String(r.Password).trim() === String(password).trim());
  if (!found) return { success: false, message: 'Invalid roll number or password' };
  return {
    success: true, role: 'student', rollNo: found.RollNo, name: found.Name,
    dept: found.Dept, year: found.Year, section: found.Section
  };
}

/**
 * Unified login used by the single login form on the index page.
 * Tries the Admin sheet first, then the Students sheet — the caller doesn't
 * need to say which role they are; the credentials themselves decide it,
 * and the page redirects based on the returned `role`.
 */
function login(identifier, password) {
  const admin = adminLogin(identifier, password);
  if (admin.success) return admin;

  const student = studentLogin(identifier, password);
  if (student.success) return student;

  return { success: false, message: 'Invalid credentials. Check your Roll Number / Username and password.' };
}

function changeStudentPassword(rollNo, oldPassword, newPassword) {
  const sh = sheet_(SHEET.STUDENTS);
  const data = sh.getDataRange().getValues();
  const headers = data[0];
  const rollCol = headers.indexOf('RollNo');
  const passCol = headers.indexOf('Password');
  for (let i = 1; i < data.length; i++) {
    if (String(data[i][rollCol]).trim().toLowerCase() === String(rollNo).trim().toLowerCase()) {
      if (String(data[i][passCol]).trim() !== String(oldPassword).trim()) {
        return { success: false, message: 'Current password is incorrect' };
      }
      sh.getRange(i + 1, passCol + 1).setValue(newPassword);
      return { success: true, message: 'Password updated successfully' };
    }
  }
  return { success: false, message: 'Student not found' };
}

/**
 * Self-service "Forgot Password" for students who are NOT logged in.
 * Verifies identity using RollNo + Name (case-insensitive match against the Students sheet),
 * then generates and stores a fresh password, returning it directly to the student.
 * No email step is required, so this works out of the box with just the Students sheet.
 */
function studentForgotPassword(rollNo, name) {
  const sh = sheet_(SHEET.STUDENTS);
  const data = sh.getDataRange().getValues();
  const headers = data[0];
  const rollCol = headers.indexOf('RollNo');
  const nameCol = headers.indexOf('Name');
  const passCol = headers.indexOf('Password');

  if (!rollNo || !name) return { success: false, message: 'Enter both Roll No and Name' };

  for (let i = 1; i < data.length; i++) {
    if (String(data[i][rollCol]).trim().toLowerCase() === String(rollNo).trim().toLowerCase()) {
      const sheetName = String(data[i][nameCol]).trim().toLowerCase();
      if (sheetName !== String(name).trim().toLowerCase()) {
        return { success: false, message: 'Roll No and Name do not match our records' };
      }
      const newPass = genPassword_(rollNo);
      sh.getRange(i + 1, passCol + 1).setValue(newPass);
      return { success: true, password: newPass, rollNo: data[i][rollCol] };
    }
  }
  return { success: false, message: 'Roll No not found. Contact your admin.' };
}

function addStudent(data) {

  if (
    !data ||
    !String(data.rollNo || '').trim() ||
    !String(data.name || '').trim() ||
    !String(data.dept || '').trim() ||
    !String(data.year || '').trim() ||
    !String(data.section || '').trim()
  ) {

    return {
      success: false,
      message:
        'All fields are mandatory: Roll No, Name, Department, Year and Section'
    };

  }


  const sh =
    sheet_(SHEET.STUDENTS);


  const existing =
    sheetToObjects_(sh);


  const rollNo =
    String(data.rollNo)
      .trim();


  const name =
    String(data.name)
      .trim();


  const dept =
    String(data.dept)
      .trim()
      .toUpperCase();


  const year =
    String(data.year)
      .trim();


  const section =
    String(data.section)
      .trim()
      .toUpperCase();


  // ============================================================
  // DUPLICATE ROLL NUMBER CHECK
  // ============================================================

  if (
    existing.some(r =>

      String(r.RollNo || '')
        .trim()
        .toLowerCase() ===

      rollNo.toLowerCase()

    )
  ) {

    return {
      success: false,
      message: 'Roll number already exists'
    };

  }


  // ============================================================
  // DEFAULT STUDENT PASSWORD
  // ============================================================

  const password =
    DEFAULT_STUDENT_PASSWORD;


  // ============================================================
  // STORE STUDENT
  // ============================================================

  sh.appendRow([

    rollNo,

    name,

    dept,

    year,

    section,

    password,

    todayStr_()

  ]);


  return {

    success: true,

    password: password,

    message:
      'Student added successfully'

  };

}

function bulkAddStudents(list) {

  if (
    !Array.isArray(list) ||
    list.length === 0
  ) {

    return [{
      success: false,
      message: 'No student data provided'
    }];

  }


  // ============================================================
  // VALIDATE ALL BULK DATA BEFORE INSERT
  // ============================================================

  for (
    let i = 0;
    i < list.length;
    i++
  ) {

    const data =
      list[i];


    if (
      !data ||
      !String(data.rollNo || '').trim() ||
      !String(data.name || '').trim() ||
      !String(data.dept || '').trim() ||
      !String(data.year || '').trim() ||
      !String(data.section || '').trim()
    ) {

      return [{

        success: false,

        message:
          'Row ' +
          (i + 1) +
          ': All fields are mandatory'

      }];

    }

  }


  const sh =
    sheet_(SHEET.STUDENTS);


  const existing =
    sheetToObjects_(sh)
      .map(r =>

        String(r.RollNo || '')
          .trim()
          .toLowerCase()

      );


  const incomingRolls = [];


  // ============================================================
  // CHECK DUPLICATES INSIDE BULK DATA
  // ============================================================

  for (
    let i = 0;
    i < list.length;
    i++
  ) {

    const roll =

      String(list[i].rollNo)
        .trim()
        .toLowerCase();


    if (
      incomingRolls.includes(roll)
    ) {

      return [{

        success: false,

        message:
          'Row ' +
          (i + 1) +
          ': Duplicate Roll No in bulk data'

      }];

    }


    incomingRolls.push(roll);

  }


  const results = [];


  // ============================================================
  // INSERT STUDENTS
  // ============================================================

  list.forEach(data => {


    const rollNo =

      String(data.rollNo)
        .trim();


    const name =

      String(data.name)
        .trim();


    const dept =

      String(data.dept)
        .trim()
        .toUpperCase();


    const year =

      String(data.year)
        .trim();


    const section =

      String(data.section)
        .trim()
        .toUpperCase();


    // ============================================================
    // EXISTING STUDENT CHECK
    // ============================================================

    if (
      existing.includes(
        rollNo.toLowerCase()
      )
    ) {

      results.push({

        rollNo: rollNo,

        success: false,

        message:
          'Already exists'

      });


      return;

    }


    // ============================================================
    // DEFAULT PASSWORD
    // ============================================================

    const password =
      DEFAULT_STUDENT_PASSWORD;


    // ============================================================
    // STORE STUDENT
    // ============================================================

    sh.appendRow([

      rollNo,

      name,

      dept,

      year,

      section,

      password,

      todayStr_()

    ]);


    existing.push(
      rollNo.toLowerCase()
    );


    results.push({

      rollNo: rollNo,

      success: true,

      password: password

    });

  });


  return results;

}

function getAllStudents() {

    rollNo: r.RollNo, name: r.Name, dept: r.Dept, year: r.Year,
    section: r.Section, password: r.Password, createdDate: fmtDate_(r.CreatedDate)
  }));
}

function deleteStudent(rollNo) {
  const sh = sheet_(SHEET.STUDENTS);
  const data = sh.getDataRange().getValues();
  for (let i = 1; i < data.length; i++) {
    if (String(data[i][0]).trim().toLowerCase() === String(rollNo).trim().toLowerCase()) {
      sh.deleteRow(i + 1);
      return { success: true };
    }
  }
  return { success: false, message: 'Not found' };
}

/**
 * A question's answer key is visible if an admin manually revealed it,
 * OR automatically once the calendar day has rolled over past its date
 * (i.e. from 12:00 AM the day after it was asked).
 */
function isRevealed_(q) {

    sheet_(SHEET.QUESTIONS);


  const data =
    sh
      .getDataRange()
      .getValues();


  const headers =
    data[0];


  const dateCol =
    headers.indexOf("Date");


  const deptCol =
    headers.indexOf("Dept");

    headers.indexOf("Year");

    headers.indexOf("Revealed");

    String(dept || "")
      .trim()
      .toUpperCase();


  const yearKey =
    String(year || "")
      .trim();

    !yearKey
  ) {

    return {

      success: false,

        "Date, Department and Year are required"

  }


  let count = 0;


  for (
    let i = 1;
    i < data.length;
    i++
  ) {


    const sameDate =

      fmtDate_(
        data[i][dateCol]
      ) === date;


    const sameDept =

      String(

      )
        .trim()

function recordViolation(data) {

  if (!data || !data.rollNo || !data.violationType) {
    return {
      success: false,
      message: 'Invalid violation data'
    };
  }

  const sh = sheet_(SHEET.VIOLATIONS);

  if (sh.getLastRow() === 0) {
    sh.appendRow([
      'ViolationID',
      'RollNo',
      'Date',
      'ViolationType',
      'ViolationLabel',
      'QID',
      'QuestionIndex',
      'DetectedAt',
      'Timestamp'
    ]);
  }

  const today = todayStr_();

  const rollNo = String(data.rollNo)
    .trim()
    .toLowerCase();

  const rows = sheetToObjects_(sh);

  // Only ONE violation per student per day
  const existing = rows.find(r =>
    String(r.RollNo).trim().toLowerCase() === rollNo &&
    fmtDate_(r.Date) === today
  );

  if (existing) {

    return {
      success: true,
      alreadyRecorded: true,
      violationType: existing.ViolationType,
      violationLabel: existing.ViolationLabel
    };

  }

  sh.appendRow([
    newId_('V'),
    data.rollNo,
    today,
    data.violationType,
    data.violationLabel || data.violationType,
    data.qid || '',
    Number(data.questionIndex || 0),
    data.detectedAt || new Date().toISOString(),
    new Date()
  ]);

  return {
    success: true,
    alreadyRecorded: false,
    violationType: data.violationType,
    violationLabel:
      data.violationLabel || data.violationType
  };
}

/**********************************************************************
 * LEADERBOARD & REPORTS
 **********************************************************************/
function periodRange_(period) {
  const now = new Date();
  const tz = Session.getScriptTimeZone() || 'Asia/Kolkata';
  let start;
  if (period === 'day') {
    start = new Date(now.getFullYear(), now.getMonth(), now.getDate());
  } else if (period === 'week') {
    const day = now.getDay();
    const diff = now.getDate() - day + (day === 0 ? -6 : 1); // Monday start
    start = new Date(now.getFullYear(), now.getMonth(), diff);
  } else if (period === 'month') {
    start = new Date(now.getFullYear(), now.getMonth(), 1);
  } else if (period === 'year') {
    start = new Date(now.getFullYear(), 0, 1);
  } else {
    start = new Date(2000, 0, 1);
  }
  return { start: Utilities.formatDate(start, tz, 'yyyy-MM-dd'), end: Utilities.formatDate(now, tz, 'yyyy-MM-dd') };
}

function getLeaderboard(
  period,
  dept,
  year,
  section
) {

  const range =
    periodRange_(
      period || "all"
    );


  const students =
    sheetToObjects_(
      sheet_(SHEET.STUDENTS)
    );


  const responses =
    sheetToObjects_(
      sheet_(SHEET.RESPONSES)
    );


  const scoreMap = {};


  responses.forEach(r => {

    const d =
      fmtDate_(r.Date);


    if (
      d < range.start ||
      d > range.end
    ) {

      return;

    }


    const key =
      String(r.RollNo)
        .trim()
        .toLowerCase();


    if (!scoreMap[key]) {

      scoreMap[key] = {

        marks: 0,

        attempted: 0,

        correct: 0

      };

    }


    scoreMap[key].marks +=
      Number(r.Marks || 0);


    scoreMap[key].attempted += 1;


    if (
      r.IsCorrect === true ||
      r.IsCorrect === "TRUE"
    ) {

      scoreMap[key].correct += 1;

    }

  });


  let list =
    students.map(s => {

      const key =
        String(s.RollNo)
          .trim()
          .toLowerCase();


      const sc =
        scoreMap[key] || {

          marks: 0,

          attempted: 0,

          correct: 0

        };


      return {

        rollNo:
          s.RollNo,

        name:
          s.Name,

        dept:
          s.Dept,

        year:
          s.Year,

        section:
          s.Section,

        marks:
          sc.marks,

        attempted:
          sc.attempted,

        correct:
          sc.correct

      };

    });


  if (dept) {

    list =
      list.filter(s =>

        String(s.dept || "")
          .trim()
          .toUpperCase() ===

        String(dept)
          .trim()
          .toUpperCase()

      );

  }


  if (year) {

    list =
      list.filter(s =>

        String(s.year) ===
        String(year)

      );

  }


  if (section) {

    list =
      list.filter(s =>

        String(s.section)
          .trim()
          .toUpperCase() ===

        String(section)
          .trim()
          .toUpperCase()

      );

  }


  list.sort(

    (a, b) =>

      b.marks -
      a.marks

      ||

      b.correct -
      a.correct

  );


  list.forEach(

    (student, index) =>

      student.rank =
        index + 1

  );


  return list;

}

function getStudentReport(rollNo, period) {
  const range = periodRange_(period || 'all');
  const responses = sheetToObjects_(sheet_(SHEET.RESPONSES))
    .filter(r => String(r.RollNo).trim().toLowerCase() === String(rollNo).trim().toLowerCase())
    .filter(r => { const d = fmtDate_(r.Date); return d >= range.start && d <= range.end; })
    .sort((a, b) => fmtDate_(a.Date).localeCompare(fmtDate_(b.Date)));

  const totalMarks = responses.reduce((s, r) => s + Number(r.Marks || 0), 0);
  const correct = responses.filter(r => r.IsCorrect === true || r.IsCorrect === 'TRUE').length;

  const leaderboard = getLeaderboard(period);
  const me = leaderboard.find(s => String(s.rollNo).trim().toLowerCase() === String(rollNo).trim().toLowerCase());

  return {
    totalAttempted: responses.length,
    correct: correct,
    totalMarks: totalMarks,
    rank: me ? me.rank : null,
    totalStudents: leaderboard.length,
    history: responses.map(r => ({
      date: fmtDate_(r.Date), qid: r.QID, selected: r.Selected,
      isCorrect: r.IsCorrect, marks: r.Marks, timeTaken: r.TimeTaken
    }))
  };
}

/**
 * Full answer-key review for a student: past questions they answered, with the
 * correct option included ONLY for questions whose answer key is currently
 * revealed (manually by admin, or automatically once the day has passed).
 */
function getAnswerKey(rollNo, period) {
  const range = periodRange_(period || 'all');
  const responses = sheetToObjects_(sheet_(SHEET.RESPONSES))
    .filter(r => String(r.RollNo).trim().toLowerCase() === String(rollNo).trim().toLowerCase())
    .filter(r => { const d = fmtDate_(r.Date); return d >= range.start && d <= range.end; });

  const questions = sheetToObjects_(sheet_(SHEET.QUESTIONS));
  const qMap = {};
  questions.forEach(q => qMap[String(q.QID)] = q);

  return responses.map(r => {
    const q = qMap[String(r.QID)];
    if (!q) return null;
    const revealed = isRevealed_(q);
    return {
      date: fmtDate_(r.Date), qid: r.QID,
      question: q.Question, optionA: q.OptionA, optionB: q.OptionB, optionC: q.OptionC, optionD: q.OptionD,

      selected: r.Selected, isCorrect: (r.IsCorrect === true || r.IsCorrect === 'TRUE'), marksScored: r.Marks,
      revealed: revealed, correctAnswer: revealed ? q.Correct : null
    };
  }).filter(Boolean).sort((a, b) => b.date.localeCompare(a.date));
}

      responses.map(r =>

        "|"

          .trim();


      return testGroups.has(
        groupKey

    });


  if (dept) {

    list =
      list.filter(s =>

        String(s.Dept || "")

          .toUpperCase() ===

          .toUpperCase()

      list.filter(s =>

        String(s.Section || "")
          .trim()
          .toUpperCase() ===

        String(section)
          .trim()
          .toUpperCase()

      );

  }


  return list

    .filter(s =>

      !attemptedRolls.has(

        String(s.RollNo)
          .trim()
          .toLowerCase()

function getDistinctYearsSections() {

  const students =
    sheetToObjects_(
      sheet_(SHEET.STUDENTS)
    );


  const departments = [

    ...new Set(

      students

        .map(s =>

          String(s.Dept || "")
            .trim()
            .toUpperCase()

        )

        .filter(Boolean)

    )

  ].sort();


  const years = [

    ...new Set(

      students

        .map(s =>

          String(s.Year || "")
            .trim()

        )

        .filter(Boolean)

    )

  ].sort(

    (a, b) =>
      Number(a) - Number(b)

  );


  const sections = [

    ...new Set(

      students

        .map(s =>

          String(s.Section || "")
            .trim()
            .toUpperCase()

        )

        .filter(Boolean)

    )

  ].sort();


  return {

    departments:
      departments,

    years:
      years,

    sections:
      sections

  };

}
